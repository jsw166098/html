# 반복문의 활용

* 가장 앞에 있는 태그 하나 가져온다.
~~~
document.quertySelector('a')
~~~

* 태그 여러개 가져온다.
~~~
document.querySelectorAll('a')
~~~

* 변수화 하여 특정 태그 가져오기
~~~
var alist = document.querySelector('a')
console.log(alist[1])
~~~

* 반복문 사용하여 확인한다.
~~~
var alist = document.querySelectorAll('a');
var i = 0;
while(i < alist.length>){
    console.log(alist[i]);
    i = i+1;
}
~~~

* 스타일 적용하여 바꾼다. 
~~~
var alist = document.querySelectorAll('a');
var i = 0;
while(i < alist.length){
    console.log(alist[i]);
    alist[i].style.color = 'powderblue';
    i = i+1;
}
~~~
